  <script>
    function homeMotor() {
      // Disable button to prevent double-clicks
      document.querySelector('.btn-home').disabled = true;
      document.querySelector('.btn-home').innerHTML = 'HOMING...';
      
      fetch('/control', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/x-www-form-urlencoded'}, 
        body: 'action=home' 
      });
      setTimeout(() => location.reload(), 5000); // Longer timeout for homing
    }

    function manualMove(direction) {
      fetch('/control', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/x-www-form-urlencoded'}, 
        body: 'action=manual&direction=' + direction 
      });
      setTimeout(() => location.reload(), 2000);
    }

    function savePosition() {
      if(confirm('Save current position as CLOSED position?')) {
        fetch('/control', { 
          method: 'POST', 
          headers: {'Content-Type': 'application/x-www-form-urlencoded'}, 
          body: 'action=save' 
        });
        setTimeout(() => location.reload(), 1000);
      }
    }

    function openCurtain() {
      // Disable button to prevent double-clicks
      event.target.disabled = true;
      event.target.innerHTML = 'OPENING...';
      
      fetch('/control', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/x-www-form-urlencoded'}, 
        body: 'action=open' 
      });
      setTimeout(() => location.reload(), 5000); // Same timeout as homing
    }

    function closeCurtain() {
      // Disable button to prevent double-clicks
      event.target.disabled = true;
      event.target.innerHTML = 'CLOSING...';
      
      fetch('/control', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/x-www-form-urlencoded'}, 
        body: 'action=close' 
      });
      setTimeout(() => location.reload(), 8000); // Longer timeout for closing
    }

    function stopMotor() {
      fetch('/control', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/x-www-form-urlencoded'}, 
        body: 'action=stop' 
      });
    }
  </script>
