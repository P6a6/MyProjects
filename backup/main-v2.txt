// Pin definitions
#define STEP_PIN 2
#define DIR_PIN 4
#define EN_PIN 16
#define HOME_SENSOR_PIN 18

// Web server
WebServer server(80);

// System state
int currentPosition = 0;        // Current position in steps from home
int closedPosition = 0;         // Fully closed position (saved in EEPROM)
bool motorRunning = false;
bool isHomed = false;

// Constants
const int HOMING_TIMEOUT_MS = 20000;     // 15 seconds timeout
const int HOMING_SPEED_DELAY = 2000;     // Very slow speed for homing (microseconds)
const int NORMAL_SPEED_DELAY = 2000;     // Much slower for normal operation
const int HEAVY_LOAD_DELAY = 3000;      // Extra slow for starting from closed position
const int MANUAL_STEP_SIZE = 300;        // Smaller steps for manual control
const int EEPROM_ADDRESS = 0;            // Address to store closed position

void setup() {
  Serial.begin(115200);
  EEPROM.begin(512);
  
  // Set pin modes
  pinMode(STEP_PIN, OUTPUT);
  pinMode(DIR_PIN, OUTPUT);
  pinMode(EN_PIN, OUTPUT);
  pinMode(HOME_SENSOR_PIN, INPUT_PULLUP);  // Pull-up for inverted sensor logic
  
  // Initialize pins
  digitalWrite(STEP_PIN, LOW);
  digitalWrite(DIR_PIN, HIGH);
  digitalWrite(EN_PIN, LOW); // Enable motor
  
  Serial.println("Smart Curtain System Starting...");
  
  // Debug sensor reading
  Serial.print("Home sensor reading: ");
  Serial.println(digitalRead(HOME_SENSOR_PIN) ? "HIGH (clear)" : "LOW (triggered)");
  
  // Load saved closed position from EEPROM
  EEPROM.get(EEPROM_ADDRESS, closedPosition);
  if (closedPosition < 0 || closedPosition > 50000) {
    closedPosition = 0; // Reset if invalid data
  }
  Serial.println("Loaded closed position: " + String(closedPosition) + " steps");
  
  // Connect to WiFi
  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi");
  
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  Serial.println("");
  Serial.println("WiFi connected!");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
  
  // Setup web server routes
  server.on("/", handleRoot);
  server.on("/control", HTTP_POST, handleControl);
  
  // Start server
  server.begin();
  Serial.println("Web server started!");
  Serial.println("Go to: http://" + WiFi.localIP().toString());
  
  // Auto-home on startup
  Serial.println("Starting auto-homing sequence...");
  autoHome();
}

void loop() {
  server.handleClient();
}

void autoHome() {
  Serial.println("AUTO-HOMING: Moving up to find home sensor...");
  
  // Debug current sensor state
  bool sensorState = digitalRead(HOME_SENSOR_PIN);
  Serial.println("Current sensor reading: " + String(sensorState ? "HIGH (clear)" : "LOW (triggered)"));
  
  // Check if already at home (sensor triggered = LOW)
  if (sensorState == LOW) {
    Serial.println("Already at home position!");
    currentPosition = 0;
    isHomed = true;
    return;
  }
  
  digitalWrite(DIR_PIN, HIGH); // Move up
  
  unsigned long startTime = millis();
  int stepCount = 0;
  
  // Move up until sensor triggers (goes LOW) or timeout
  while (digitalRead(HOME_SENSOR_PIN) == HIGH) {
    // Check timeout
    if (millis() - startTime > HOMING_TIMEOUT_MS) {
      Serial.println("HOMING TIMEOUT! Stopping motor.");
      return;
    }
    
    // Take one step
    digitalWrite(STEP_PIN, HIGH);
    delayMicroseconds(50);
    digitalWrite(STEP_PIN, LOW);
    delayMicroseconds(HOMING_SPEED_DELAY);
    
    stepCount++;
    
    // Debug sensor reading every 200 steps
    if (stepCount % 200 == 0) {
      Serial.println("Homing step " + String(stepCount) + ", Sensor: " + String(digitalRead(HOME_SENSOR_PIN) ? "HIGH (clear)" : "LOW (triggered)"));
    }
    
    // Handle web requests occasionally
    if (stepCount % 100 == 0) {
      server.handleClient();
    }
  }
  
  // Home sensor triggered!
  Serial.println("HOME DETECTED! Stopped at home position.");
  currentPosition = 0;
  isHomed = true;
}

void handleControl() {
  if (!server.hasArg("action")) {
    server.send(400, "text/plain", "No action specified");
    return;
  }
  
  String action = server.arg("action");
  Serial.println("Command received: " + action);
  
  if (action == "stop") {
    motorRunning = false;
    Serial.println("EMERGENCY STOP");
  }
  else if (action == "home") {
    Serial.println("Starting homing sequence...");
    autoHome();
  }
  else if (action == "manual" && server.hasArg("direction")) {
    String direction = server.arg("direction");
    Serial.println("Manual move: " + direction);
    
    // Choose speed based on current position
    int speedDelay = NORMAL_SPEED_DELAY;
    if (currentPosition >= (closedPosition * 0.8) && direction == "up") {
      speedDelay = HEAVY_LOAD_DELAY;
      Serial.println("Using HEAVY LOAD speed for manual UP from closed position");
    }
    
    if (direction == "up") {
      digitalWrite(DIR_PIN, HIGH);
      runMotorSteps(MANUAL_STEP_SIZE, speedDelay, true, true);
      currentPosition = max(0, currentPosition - MANUAL_STEP_SIZE);
    } else if (direction == "down") {
      digitalWrite(DIR_PIN, LOW);
      runMotorSteps(MANUAL_STEP_SIZE, NORMAL_SPEED_DELAY, false, false);
      currentPosition += MANUAL_STEP_SIZE;
    }
  }
  else if (action == "save") {
    if (currentPosition > 50) { // Prevent saving home as closed position
      closedPosition = currentPosition;
      EEPROM.put(EEPROM_ADDRESS, closedPosition);
      EEPROM.commit();
      Serial.println("Saved closed position: " + String(closedPosition) + " steps");
    } else {
      Serial.println("ERROR: Cannot save closed position - too close to home!");
    }
  }
  else if (action == "open") {
    Serial.println("Opening curtain - going to home position");
    moveToPosition(0); // Go to home
  }
  else if (action == "close" && closedPosition > 0) {
    Serial.println("Closing curtain - going to closed position");
    moveToPosition(closedPosition); // Go to closed position
  }
  else {
    Serial.println("Unknown or invalid command: " + action);
  }
  
  server.send(200, "text/plain", "OK");
}

void moveToPosition(int targetPosition) {
  if (targetPosition == currentPosition) {
    Serial.println("Already at target position");
    return;
  }
  
  int stepsToMove = abs(targetPosition - currentPosition);
  bool moveUp = (targetPosition < currentPosition);
  
  Serial.println("Moving to position " + String(targetPosition) + " (currently at " + String(currentPosition) + ")");
  Serial.println("Steps to move: " + String(stepsToMove) + ", Direction: " + String(moveUp ? "UP" : "DOWN"));
  
  // Use slower speed if starting from closed position (heavy load)
  int speedDelay = NORMAL_SPEED_DELAY;
  if (currentPosition >= (closedPosition * 0.8) && moveUp) {
    speedDelay = HEAVY_LOAD_DELAY;
    Serial.println("Using HEAVY LOAD speed (extra slow) - starting from closed position");
  }
  
  digitalWrite(DIR_PIN, moveUp ? HIGH : LOW);
  runMotorSteps(stepsToMove, speedDelay, moveUp, moveUp);
  
  currentPosition = targetPosition;
  Serial.println("Movement complete. New position: " + String(currentPosition));
}

void runMotorSteps(int steps, int delayMicros, bool direction, bool checkSensor) {
  digitalWrite(DIR_PIN, direction ? HIGH : LOW);
  motorRunning = true;
  
  Serial.println("Running " + String(steps) + " steps, direction: " + String(direction ? "UP" : "DOWN"));
  Serial.println("Speed: " + String(delayMicros) + " microseconds delay");
  
  for(int i = 0; i < steps && motorRunning; i++) {
    // Check home sensor if moving up (sensor triggered = LOW)
    if (checkSensor && digitalRead(HOME_SENSOR_PIN) == LOW) {
      Serial.println("Home sensor triggered during movement - stopping");
      Serial.println("Sensor reading: LOW (triggered)");
      currentPosition = 0;
      break;
    }
    
    // Debug sensor every 100 steps when checking
    if (checkSensor && i % 100 == 0) {
      Serial.println("Step " + String(i) + ", Sensor: " + String(digitalRead(HOME_SENSOR_PIN) ? "HIGH (clear)" : "LOW (triggered)"));
    }
    
    digitalWrite(STEP_PIN, HIGH);
    delayMicroseconds(50);
    digitalWrite(STEP_PIN, LOW);
    delayMicroseconds(delayMicros);
    
    // Handle web requests every 100 steps
    if(i % 100 == 0) {
      server.handleClient();
    }
  }
  
  motorRunning = false;
  Serial.println("Motor movement finished");
}
