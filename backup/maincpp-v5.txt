// Schedule structure
struct Schedule {

int minute;
bool isPM;
byte days;

// Last execution tracking
struct LastExecution {
int day;
int hour;
int minute;
};

void setup() {

EEPROM.begin(512);

if (!SPIFFS.begin(true)) {
Serial.println("SPIFFS Mount Failed");
return;
}

pinMode(STEP_PIN, OUTPUT);

digitalWrite(DIR_PIN, HIGH);
digitalWrite(EN_PIN, LOW);

Serial.println("Smart Curtain System Starting");

EEPROM.get(EEPROM_ADDRESS, closedPosition);
if (closedPosition < 0 || closedPosition > 50000) {
closedPosition = 0;
}
Serial.println("Loaded closed position: " + String(closedPosition) + " steps");

loadLastExecutionTimes();

WiFi.begin(ssid, password);
Serial.print("Connecting to WiFi");
while (WiFi.status() != WL_CONNECTED) {
delay(500);
Serial.print(".");
}

Serial.print("IP address: ");

server.on("/control", HTTP_POST, handleControl);

// NEW: Siri control routes
server.on("/curtain/up", HTTP_GET, handleSiriUp);
server.on("/curtain/down", HTTP_GET, handleSiriDown);

server.onNotFound([]() {
String path = server.uri();
if (path.endsWith("/")) path += "index.html";

String contentType = "text/plain";
if (path.endsWith(".html")) contentType = "text/html";
else if (path.endsWith(".css")) contentType = "text/css";
else if (path.endsWith(".js")) contentType = "application/javascript";

if (SPIFFS.exists(path)) {
  File file = SPIFFS.open(path, "r");
  server.streamFile(file, contentType);
  file.close();
} else {
  server.send(404, "text/plain", "File not found");
}
});

server.begin();
Serial.println("Web server started!");
Serial.println("Siri Control URLs:");
Serial.println("Curtain Up: http://" + WiFi.localIP().toString() + "/curtain/up");
Serial.println("Curtain Down: http://" + WiFi.localIP().toString() + "/curtain/down");

Serial.println("Starting auto-homing sequence");
autoHome();
}

// NEW: Siri "curtain up" handler
void handleSiriUp() {
Serial.println("*** SIRI COMMAND: CURTAIN UP ***");

if (motorRunning) {
Serial.println("Motor already running - ignoring Siri up command");
server.send(200, "text/plain", "Motor busy - curtain already moving");
return;
}

Serial.println("Opening curtain via Siri command");
autoHome();
positionLost = false;

server.send(200, "text/plain", "Curtain opening");
}

// NEW: Siri "curtain down" handler  
void handleSiriDown() {
Serial.println("*** SIRI COMMAND: CURTAIN DOWN ***");

if (closedPosition == 0) {
Serial.println("No closed position set - cannot close curtain");
server.send(400, "text/plain", "No closed position set");
return;
}

if (motorRunning) {
Serial.println("Motor already running - ignoring Siri down command");
server.send(200, "text/plain", "Motor busy - curtain already moving");
return;
}

if (positionLost) {
Serial.println("Position lost - homing first before closing");
autoHome();
delay(1000); // Give homing time to complete
if (!positionLost && closedPosition > 0) {
moveToPosition(closedPosition);
}
server.send(200, "text/plain", "Position was lost - homed and closing curtain");
} else {
Serial.println("Closing curtain via Siri command");
moveToPosition(closedPosition);
server.send(200, "text/plain", "Curtain closing");
}
}

void loop() {
server.handleClient();

if (!motorRunning && millis() - lastScheduleCheck > 30000) {
checkSchedules();
lastScheduleCheck = millis();
}
}

void setupTime() {
configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
Serial.println("Waiting for time synchronization…");

int attempts = 0;
while (time(nullptr) < 8 * 3600 * 2 && attempts < 15) {
delay(1000);
Serial.print(".");
attempts++;
}

if (time(nullptr) < 8 * 3600 * 2) {
Serial.println("\nFailed to sync time");
} else {
Serial.println("\nTime synchronized successfully");
struct tm timeinfo;
if (getLocalTime(&timeinfo)) {
Serial.printf("Current time: %04d-%02d-%02d %02d:%02d:%02d\n",
timeinfo.tm_year + 1900, timeinfo.tm_mon + 1, timeinfo.tm_mday,
timeinfo.tm_hour, timeinfo.tm_min, timeinfo.tm_sec);
}
}
}

void checkSchedules() {
struct tm timeinfo;
if (!getLocalTime(&timeinfo)) {
return;
}

for (int i = 0; i < scheduleCount; i++) {
if (shouldExecuteSchedule(i, &timeinfo)) {
Serial.printf("*** EXECUTING SCHEDULE %d ***\n", i);

  lastExecutions[i].day = timeinfo.tm_yday + 1;
  lastExecutions[i].hour = timeinfo.tm_hour;
  lastExecutions[i].minute = timeinfo.tm_min;
  
  saveLastExecutionTimes();
  executeScheduledAction(schedules[i].action);
  
  Serial.printf("Schedule %d execution completed\n", i);
}
}
}

bool shouldExecuteSchedule(int scheduleIndex, struct tm* timeinfo) {
if (scheduleIndex >= scheduleCount) return false;

Schedule& sched = schedules[scheduleIndex];
LastExecution& lastExec = lastExecutions[scheduleIndex];

int schedHour24 = sched.hour;
if (sched.isPM && sched.hour != 12) {
schedHour24 += 12;
} else if (!sched.isPM && sched.hour == 12) {
schedHour24 = 0;
}

int currentDay = timeinfo->tm_wday;
if (!(sched.days & (1 << currentDay))) {
return false;
}

if (timeinfo->tm_hour != schedHour24) {
return false;
}

int timeDiff = abs(timeinfo->tm_min - sched.minute);
if (timeDiff > 2 && timeDiff < 58) {
return false;
}

int currentDay365 = timeinfo->tm_yday + 1;

if (lastExec.day == currentDay365 &&
lastExec.hour == timeinfo->tm_hour &&
abs(lastExec.minute - timeinfo->tm_min) <= 2) {
return false;
}

return true;
}

void executeScheduledAction(const char* action) {
if (motorRunning) {
Serial.println("Motor busy - skipping scheduled action");
return;
}

if (strcmp(action, "open") == 0) {
Serial.println("Executing scheduled OPEN");
autoHome();
} else if (strcmp(action, "close") == 0 && closedPosition > 0) {
if (positionLost) {
Serial.println("Position lost - homing first");
autoHome();
} else {
Serial.println("Executing scheduled CLOSE");
moveToPosition(closedPosition);
}
}
}

void saveSchedulesToEEPROM() {
EEPROM.put(SCHEDULE_EEPROM_ADDRESS, scheduleCount);
EEPROM.put(SCHEDULE_EEPROM_ADDRESS + sizeof(int), schedules);
EEPROM.commit();
Serial.println("Saved " + String(scheduleCount) + " schedules to EEPROM");
}

void loadSchedulesFromEEPROM() {
EEPROM.get(SCHEDULE_EEPROM_ADDRESS, scheduleCount);

if (scheduleCount < 0 || scheduleCount > MAX_SCHEDULES) {
scheduleCount = 0;
Serial.println("No valid schedules found in EEPROM");
return;
}

EEPROM.get(SCHEDULE_EEPROM_ADDRESS + sizeof(int), schedules);
Serial.println("Loaded " + String(scheduleCount) + " schedules from EEPROM");
}

void saveLastExecutionTimes() {
EEPROM.put(EXECUTION_EEPROM_ADDRESS, lastExecutions);
EEPROM.commit();
}

void loadLastExecutionTimes() {
for (int i = 0; i < MAX_SCHEDULES; i++) {
lastExecutions[i].day = -1;
lastExecutions[i].hour = -1;
lastExecutions[i].minute = -1;
}
EEPROM.get(EXECUTION_EEPROM_ADDRESS, lastExecutions);
}

void autoHome() {
Serial.println("=== STARTING HOMING ===");

motorRunning = true;
delay(500);

bool initialSensor = digitalRead(HOME_SENSOR_PIN);
Serial.println("Initial sensor state: " + String(initialSensor ? "HIGH" : "LOW"));

if (initialSensor == HIGH) {
Serial.println("Already at home!");
currentPosition = 0;
isHomed = true;
positionLost = false;
motorRunning = false;
return;
}

Serial.println("Moving up to find home sensor");
digitalWrite(DIR_PIN, HIGH);

unsigned long startTime = millis();
int stepCount = 0;
int consecutiveHighReadings = 0;
const int SENSOR_CONFIRM_COUNT = 5;

while (true) {
if (millis() - startTime > HOMING_TIMEOUT_MS) {
Serial.println("=== TIMEOUT REACHED ===");
positionLost = true;
motorRunning = false;
return;
}

digitalWrite(STEP_PIN, HIGH);
delayMicroseconds(50);
digitalWrite(STEP_PIN, LOW);
delayMicroseconds(HOMING_SPEED_DELAY);

stepCount++;

bool sensorReading = digitalRead(HOME_SENSOR_PIN);

if (sensorReading == HIGH) {
  consecutiveHighReadings++;
  if (consecutiveHighReadings >= SENSOR_CONFIRM_COUNT) {
    Serial.println("=== HOME SENSOR CONFIRMED ===");
    currentPosition = 0;
    isHomed = true;
    positionLost = false;
    motorRunning = false;
    return;
  }
} else {
  consecutiveHighReadings = 0;
}

if (stepCount % 500 == 0) {
  Serial.println("Step " + String(stepCount));
}
}
}

void handleRoot() {
File file = SPIFFS.open("/index.html", "r");
if (!file) {
server.send(404, "text/plain", "File not found");
return;
}

String html = file.readString();
file.close();

String systemStatus = isHomed ? "Ready" : "Not Homed";

String curtainPosition;
if (closedPosition > 0) {
int percentage = max(0, min(100, (int)round(((float)(closedPosition - currentPosition) / closedPosition) * 100)));
curtainPosition = String(percentage) + "% Open";
} else {
curtainPosition = "No Closed Position Set";
}

String connectionStatus = positionLost ? "Position Lost" : "Connected";
String scheduleListHTML = generateScheduleListHTML();

html.replace("{SYSTEM_STATUS}", systemStatus);
html.replace("{CURTAIN_POSITION}", curtainPosition);
html.replace("{CONNECTION_STATUS}", connectionStatus);
html.replace("{SCHEDULE_LIST}", scheduleListHTML);

server.send(200, "text/html", html);
}

String generateScheduleListHTML() {
String html = "";

for (int i = 0; i < scheduleCount; i++) {
String timeStr = String(schedules[i].hour) + ":";
if (schedules[i].minute < 10) timeStr += "0";
timeStr += String(schedules[i].minute) + " ";
timeStr += schedules[i].isPM ? "PM" : "AM";

String daysStr = "";
const char* dayNames[] = {"Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"};
int dayCount = 0;

for (int d = 0; d < 7; d++) {
  if (schedules[i].days & (1 << d)) {
    if (dayCount > 0) daysStr += ", ";
    daysStr += dayNames[d];
    dayCount++;
  }
}

if (dayCount == 7) {
  daysStr = "Every Day";
}

String actionStr = (strcmp(schedules[i].action, "open") == 0) ? "Open Curtain" : "Close Curtain";

html += "<div class=\"schedule-item\">";
html += "<div class=\"schedule-details\">";
html += "<div class=\"schedule-time\">" + timeStr + "</div>";
html += "<div class=\"schedule-days\">" + daysStr + "</div>";
html += "<div class=\"schedule-action\">Action: " + actionStr + "</div>";
html += "</div>";
html += "<button class=\"btn-delete\" onclick=\"deleteSchedule(" + String(i) + ")\">DELETE</button>";
html += "</div>";
}

if (scheduleCount == 0) {
html = "<div style=\"text-align:center;color:#666;padding:20px;\">No schedules configured</div>";
}

return html;
}

void handleControl() {
if (!server.hasArg("action")) {
server.send(400, "text/plain", "No action specified");
return;
}

String action = server.arg("action");
Serial.println("Command received: " + action);

if (action == "stop") {
motorRunning = false;
positionLost = true;
Serial.println("EMERGENCY STOP");
}
else if (action == "home") {
if (motorRunning) {
Serial.println("Stopping current operation before homing");
motorRunning = false;
delay(200);
}
Serial.println("Starting homing sequence");
autoHome();
positionLost = false;
}
else if (action == "manual" && server.hasArg("direction")) {
if (motorRunning) {
Serial.println("Motor already running - ignoring manual command");
server.send(200, "text/plain", "Motor busy");
return;
}

String direction = server.arg("direction");
Serial.println("Manual move: " + direction);

int speedDelay = NORMAL_SPEED_DELAY;
if (currentPosition >= (closedPosition * 0.8) && direction == "up") {
  speedDelay = HEAVY_LOAD_DELAY;
}

if (direction == "up") {
  digitalWrite(DIR_PIN, HIGH);
  runMotorSteps(MANUAL_STEP_SIZE, speedDelay, true, true);
  currentPosition = max(0, currentPosition - MANUAL_STEP_SIZE);
} else if (direction == "down") {
  digitalWrite(DIR_PIN, LOW);
  runMotorSteps(MANUAL_STEP_SIZE, NORMAL_SPEED_DELAY, false, false);
  currentPosition += MANUAL_STEP_SIZE;
}
}
else if (action == "save") {
if (currentPosition > 50) {
closedPosition = currentPosition;
EEPROM.put(EEPROM_ADDRESS, closedPosition);
EEPROM.commit();
Serial.println("Saved closed position: " + String(closedPosition) + " steps");
} else {
Serial.println("ERROR: Cannot save closed position - too close to home!");
}
}
else if (action == "open") {
if (motorRunning) {
Serial.println("Motor already running - ignoring open command");
server.send(200, "text/plain", "Motor busy");
return;
}
Serial.println("Opening curtain");
autoHome();
positionLost = false;
}
else if (action == "close" && closedPosition > 0) {
if (motorRunning) {
Serial.println("Motor already running - ignoring close command");
server.send(200, "text/plain", "Motor busy");
return;
}
if (positionLost) {
Serial.println("Cannot close - position lost after emergency stop");
} else {
Serial.println("Closing curtain");
moveToPosition(closedPosition);
}
}
else if (action == "add_schedule") {
Serial.println("*** ADDING NEW SCHEDULE ***");

if (scheduleCount >= MAX_SCHEDULES) {
  Serial.println("Maximum schedules reached");
  server.send(400, "text/plain", "Maximum schedules reached");
  return;
}

if (!server.hasArg("hour") || !server.hasArg("minute") || !server.hasArg("ampm") || 
    !server.hasArg("days") || !server.hasArg("schedule_action")) {
  Serial.println("Missing parameters");
  server.send(400, "text/plain", "Missing parameters");
  return;
}

Schedule newSchedule;
newSchedule.hour = server.arg("hour").toInt();
newSchedule.minute = server.arg("minute").toInt();
newSchedule.isPM = (server.arg("ampm") == "PM");

String daysStr = server.arg("days");
newSchedule.days = 0;

for (int i = 0; i < daysStr.length(); i++) {
  if (daysStr.charAt(i) >= '0' && daysStr.charAt(i) <= '6') {
    int day = daysStr.charAt(i) - '0';
    newSchedule.days |= (1 << day);
  }
}

String scheduleAction = server.arg("schedule_action");
if (scheduleAction == "open") {
  strcpy(newSchedule.action, "open");
} else {
  strcpy(newSchedule.action, "close");
}

schedules[scheduleCount] = newSchedule;

lastExecutions[scheduleCount].day = -1;
lastExecutions[scheduleCount].hour = -1;
lastExecutions[scheduleCount].minute = -1;

scheduleCount++;

saveSchedulesToEEPROM();
saveLastExecutionTimes();

Serial.println("*** SCHEDULE SAVED SUCCESSFULLY ***");
Serial.printf("Schedule: %s at %d:%02d %s\n", 
             newSchedule.action, newSchedule.hour, newSchedule.minute, 
             newSchedule.isPM ? "PM" : "AM");
}
else if (action == "delete_schedule" && server.hasArg("index")) {
int index = server.arg("index").toInt();
if (index >= 0 && index < scheduleCount) {
for (int i = index; i < scheduleCount - 1; i++) {
schedules[i] = schedules[i + 1];
lastExecutions[i] = lastExecutions[i + 1];
}
scheduleCount--;
saveSchedulesToEEPROM();
saveLastExecutionTimes();

  Serial.println("Deleted schedule at index " + String(index));
}
}

server.send(200, "text/plain", "OK");
}

void moveToPosition(int targetPosition) {
if (targetPosition == currentPosition) {
Serial.println("Already at target position");
return;
}

int stepsToMove = abs(targetPosition - currentPosition);
bool moveUp = (targetPosition < currentPosition);

Serial.println("Moving to position " + String(targetPosition));

int speedDelay = NORMAL_SPEED_DELAY;
if (currentPosition >= (closedPosition * 0.8) && moveUp) {
speedDelay = HEAVY_LOAD_DELAY;
}

digitalWrite(DIR_PIN, moveUp ? HIGH : LOW);
runMotorSteps(stepsToMove, speedDelay, moveUp, moveUp);

currentPosition = targetPosition;
Serial.println("Movement complete. New position: " + String(currentPosition));
}

void runMotorSteps(int steps, int delayMicros, bool direction, bool checkSensor) {
digitalWrite(DIR_PIN, direction ? HIGH : LOW);
motorRunning = true;

Serial.println("Running " + String(steps) + " steps, direction: " + String(direction ? "UP" : "DOWN"));

int consecutiveHighReadings = 0;
const int SENSOR_CONFIRM_COUNT = 3;

for(int i = 0; i < steps && motorRunning; i++) {
if (i % 10 == 0) {
server.handleClient();
if (!motorRunning) {
Serial.println("Emergency stop detected during movement");
break;
}
}

if (checkSensor && direction) {
  bool sensorReading = digitalRead(HOME_SENSOR_PIN);
  
  if (sensorReading == HIGH) {
    consecutiveHighReadings++;
    if (consecutiveHighReadings >= SENSOR_CONFIRM_COUNT) {
      Serial.println("Home sensor confirmed during movement - stopping");
      currentPosition = 0;
      positionLost = false;
      break;
    }
  } else {
    consecutiveHighReadings = 0;
  }
}

digitalWrite(STEP_PIN, HIGH);
delayMicroseconds(50);
digitalWrite(STEP_PIN, LOW);
delayMicroseconds(delayMicros);
}

motorRunning = false;
Serial.println("Motor movement finished");
}
