WebServer server(80);

bool positionLost = false;      // Track if position is unreliable after emergency stop

// Schedule structure
struct Schedule {
  bool enabled;
  int hour;        // 1-12
  int minute;      // 0, 15, 30, 45
  bool isPM;       // true for PM, false for AM
  byte days;       // Bit field: bit 0=Sunday, bit 1=Monday, etc.
  String action;   // "open" or "close"
};

const int MAX_SCHEDULES = 10;
Schedule schedules[MAX_SCHEDULES];
int scheduleCount = 0;

// Constants
const int HOMING_TIMEOUT_MS = 90000;     // 90 seconds for extreme power loss recovery
const int HOMING_SPEED_DELAY = 2000;     // Very slow speed for homing (microseconds)
const int NORMAL_SPEED_DELAY = 2000;     // Much slower for normal operation
const int HEAVY_LOAD_DELAY = 3000;      // Extra slow for starting from closed position
const int MANUAL_STEP_SIZE = 300;        // Smaller steps for manual control
const int EEPROM_ADDRESS = 0;            // Address to store closed position
const int SCHEDULE_EEPROM_ADDRESS = 100; // Address to store schedules

void setup() {
  Serial.begin(115200);
  EEPROM.begin(512);

  // Initialize SPIFFS
  if (!SPIFFS.begin(true)) {
    Serial.println("SPIFFS Mount Failed");
    return;
  } else {
    Serial.println("SPIFFS Mount Successful");
    if (SPIFFS.exists("/index.html")) {
      Serial.println("index.html found!");
    } else {
      Serial.println("index.html NOT found!");
    }
  }

  // Set pin modes
  pinMode(STEP_PIN, OUTPUT);
  pinMode(DIR_PIN, OUTPUT);
  pinMode(EN_PIN, OUTPUT);
  pinMode(HOME_SENSOR_PIN, INPUT_PULLUP);  // Pull-up for sensor

  // Initialize pins
  digitalWrite(STEP_PIN, LOW);
  digitalWrite(DIR_PIN, HIGH);
  digitalWrite(EN_PIN, LOW); // Enable motor

  Serial.println("Smart Curtain System Starting…");

  // Debug sensor reading
  Serial.print("Home sensor reading: ");
  Serial.println(digitalRead(HOME_SENSOR_PIN) ? "HIGH (triggered)" : "LOW (clear)");

  // Load saved closed position from EEPROM
  EEPROM.get(EEPROM_ADDRESS, closedPosition);
  if (closedPosition < 0 || closedPosition > 50000) {
    closedPosition = 0; // Reset if invalid data
  }
  Serial.println("Loaded closed position: " + String(closedPosition) + " steps");

  // Load schedules from EEPROM
  loadSchedulesFromEEPROM();

  // Connect to WiFi
  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi");

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("");
  Serial.println("WiFi connected!");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());

  // Setup time synchronization
  setupTime();

  // Setup web server routes
  server.on("/", handleRoot);
  server.on("/control", HTTP_POST, handleControl);
  
  // Serve static files from SPIFFS
  server.onNotFound([]() {
    String path = server.uri();
    if (path.endsWith("/")) path += "index.html";
    
    String contentType = "text/plain";
    if (path.endsWith(".html")) contentType = "text/html";
    else if (path.endsWith(".css")) contentType = "text/css";
    else if (path.endsWith(".js")) contentType = "application/javascript";
    
    if (SPIFFS.exists(path)) {
      File file = SPIFFS.open(path, "r");
      server.streamFile(file, contentType);
      file.close();
    } else {
      server.send(404, "text/plain", "File not found");
    }
  });

  // Start server
  server.begin();
  Serial.println("Web server started!");
  Serial.println("Go to: http://" + WiFi.localIP().toString());

  // Auto-home on startup
  Serial.println("Starting auto-homing sequence…");
  autoHome();
}

void loop() {
  server.handleClient();
  
  // Check schedules every minute (when not running motor operations)
  if (!motorRunning && millis() - lastScheduleCheck > 60000) {
    checkSchedules();
    lastScheduleCheck = millis();
  }
}

void setupTime() {
  configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
  Serial.println("Waiting for time synchronization...");
  
  int attempts = 0;
  while (time(nullptr) < 8 * 3600 * 2 && attempts < 10) {
    delay(1000);
    Serial.print(".");
    attempts++;
  }
  
  if (time(nullptr) < 8 * 3600 * 2) {
    Serial.println("\nFailed to sync time - schedules will not work properly");
  } else {
    Serial.println("\nTime synchronized successfully");
    
    // Print current time
    struct tm timeinfo;
    if (getLocalTime(&timeinfo)) {
      Serial.printf("Current time: %02d:%02d:%02d %s\n", 
                   timeinfo.tm_hour > 12 ? timeinfo.tm_hour - 12 : (timeinfo.tm_hour == 0 ? 12 : timeinfo.tm_hour),
                   timeinfo.tm_min,
                   timeinfo.tm_sec,
                   timeinfo.tm_hour >= 12 ? "PM" : "AM");
    }
  }
}

void checkSchedules() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    return; // Can't check schedules without valid time
  }
  
  int currentHour = timeinfo.tm_hour;
  int currentMinute = timeinfo.tm_min;
  int currentDay = timeinfo.tm_wday; // 0=Sunday, 1=Monday, etc.
  
  // Convert to 12-hour format
  bool currentIsPM = currentHour >= 12;
  if (currentHour > 12) currentHour -= 12;
  if (currentHour == 0) currentHour = 12;
  
  // Round current minute to nearest 15-minute interval for checking
  int roundedMinute = (currentMinute / 15) * 15;
  
  for (int i = 0; i < scheduleCount; i++) {
    if (!schedules[i].enabled) continue;
    
    // Check if this schedule matches current time
    if (schedules[i].hour == currentHour && 
        schedules[i].minute == roundedMinute && 
        schedules[i].isPM == currentIsPM &&
        (schedules[i].days & (1 << currentDay))) {
      
      // Execute the scheduled action
      Serial.printf("Executing scheduled action: %s at %02d:%02d %s\n",
                   schedules[i].action.c_str(),
                   schedules[i].hour,
                   schedules[i].minute,
                   schedules[i].isPM ? "PM" : "AM");
      
      executeScheduledAction(schedules[i].action);
      
      // Disable this schedule temporarily to prevent multiple executions
      schedules[i].enabled = false;
      
      // Re-enable after 2 minutes
      // Note: In a real implementation, you'd want a more sophisticated way to handle this
      break; // Only execute one schedule per check
    }
  }
  
  // Re-enable all schedules if we're past their execution time
  for (int i = 0; i < scheduleCount; i++) {
    if (!schedules[i].enabled) {
      // Simple re-enable logic - re-enable if current minute is different
      if (roundedMinute != schedules[i].minute) {
        schedules[i].enabled = true;
      }
    }
  }
}

void executeScheduledAction(String action) {
  if (motorRunning) {
    Serial.println("Motor busy - skipping scheduled action");
    return;
  }
  
  if (action == "open") {
    Serial.println("Executing scheduled OPEN");
    autoHome(); // Use reliable homing for open
  } else if (action == "close" && closedPosition > 0) {
    if (positionLost) {
      Serial.println("Position lost - cannot execute scheduled close. Homing first...");
      autoHome();
    } else {
      Serial.println("Executing scheduled CLOSE");
      moveToPosition(closedPosition);
    }
  }
}

void autoHome() {
  Serial.println("=== STARTING RELIABLE HOMING ===");
  
  // COMPLETELY stop all other operations
  motorRunning = false;
  delay(500); // Give everything time to stop
  
  // Check sensor first
  bool initialSensor = digitalRead(HOME_SENSOR_PIN);
  Serial.println("Initial sensor state: " + String(initialSensor ? "HIGH" : "LOW"));
  
  if (initialSensor == HIGH) {
    Serial.println("Already at home!");
    currentPosition = 0;
    isHomed = true;
    positionLost = false;
    return;
  }
  
  Serial.println("Moving up to find home sensor - NO INTERRUPTIONS ALLOWED");
  
  // Set direction
  digitalWrite(DIR_PIN, HIGH); // Move up
  
  unsigned long startTime = millis();
  int stepCount = 0;
  int consecutiveHighReadings = 0; // For sensor filtering
  const int SENSOR_CONFIRM_COUNT = 5; // Need 5 consecutive HIGH readings
  
  // SIMPLE LOOP - No web handling, no interruptions except timeout
  while (true) {
    // ONLY check timeout
    if (millis() - startTime > HOMING_TIMEOUT_MS) {
      Serial.println("=== TIMEOUT REACHED ===");
      Serial.println("Steps taken: " + String(stepCount));
      Serial.println("Time elapsed: " + String((millis() - startTime)/1000) + " seconds");
      Serial.println("Final sensor: " + String(digitalRead(HOME_SENSOR_PIN) ? "HIGH" : "LOW"));
      positionLost = true;
      return;
    }
    
    // Take one step
    digitalWrite(STEP_PIN, HIGH);
    delayMicroseconds(50);
    digitalWrite(STEP_PIN, LOW);
    delayMicroseconds(HOMING_SPEED_DELAY);
    
    stepCount++;
    
    // Check sensor with filtering to avoid noise
    bool sensorReading = digitalRead(HOME_SENSOR_PIN);
    
    if (sensorReading == HIGH) {
      consecutiveHighReadings++;
      if (consecutiveHighReadings >= SENSOR_CONFIRM_COUNT) {
        Serial.println("=== HOME SENSOR CONFIRMED ===");
        Serial.println("Steps taken: " + String(stepCount));
        Serial.println("Consecutive HIGH readings: " + String(consecutiveHighReadings));
        currentPosition = 0;
        isHomed = true;
        positionLost = false;
        return;
      }
    } else {
      consecutiveHighReadings = 0; // Reset if we get a LOW reading
    }
    
    // Status every 500 steps
    if (stepCount % 500 == 0) {
      Serial.println("Step " + String(stepCount) + ", Sensor: " + String(sensorReading ? "HIGH" : "LOW") + ", Consecutive: " + String(consecutiveHighReadings));
    }
  }
}

void handleRoot() {
  // Serve the main HTML file from SPIFFS
  File file = SPIFFS.open("/index.html", "r");
  if (!file) {
    server.send(404, "text/plain", "File not found");
    return;
  }
  
  // Read the HTML template
  String html = file.readString();
  file.close();
  
  // Calculate real status values
  String systemStatus;
  String systemStatusClass = "status-value";
  if (isHomed) {
    systemStatus = "Ready";
  } else {
    systemStatus = "Not Homed";
    systemStatusClass = "status-value warning";
  }
  
  // Calculate curtain position percentage
  String curtainPosition;
  if (closedPosition > 0) {
    // currentPosition = 0 means fully open (at home sensor)
    // currentPosition = closedPosition means fully closed
    // So percentage open = (closedPosition - currentPosition) / closedPosition * 100
    int percentage = max(0, min(100, (int)round(((float)(closedPosition - currentPosition) / closedPosition) * 100)));
    curtainPosition = String(percentage) + "% Open";
  } else {
    curtainPosition = "No Closed Position Set";
  }
  
  // Connection status based on position reliability
  String connectionStatus;
  String connectionStatusClass = "status-value";
  if (positionLost) {
    connectionStatus = "Position Lost";
    connectionStatusClass = "status-value error";
  } else {
    connectionStatus = "Connected";
  }
  
  // Generate schedule list HTML
  String scheduleListHTML = generateScheduleListHTML();
  
  // Replace placeholders with current system state
  html.replace("{SYSTEM_STATUS}", systemStatus);
  html.replace("{CURTAIN_POSITION}", curtainPosition);
  html.replace("{CONNECTION_STATUS}", connectionStatus);
  html.replace("{SCHEDULE_LIST}", scheduleListHTML);
  
  server.send(200, "text/html", html);
}

String generateScheduleListHTML() {
  String html = "";
  
  for (int i = 0; i < scheduleCount; i++) {
    if (!schedules[i].enabled && schedules[i].hour == 0) continue; // Skip empty slots
    
    // Format time
    String timeStr = String(schedules[i].hour) + ":";
    if (schedules[i].minute < 10) timeStr += "0";
    timeStr += String(schedules[i].minute) + " ";
    timeStr += schedules[i].isPM ? "PM" : "AM";
    
    // Format days
    String daysStr = "";
    const char* dayNames[] = {"Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"};
    int dayCount = 0;
    
    for (int d = 0; d < 7; d++) {
      if (schedules[i].days & (1 << d)) {
        if (dayCount > 0) daysStr += ", ";
        daysStr += dayNames[d];
        dayCount++;
      }
    }
    
    if (dayCount == 7) {
      daysStr = "Every Day";
    }
    
    // Format action
    String actionStr = schedules[i].action == "open" ? "Open Curtain" : "Close Curtain";
    
    html += "<div class=\"schedule-item\">";
    html += "<div class=\"schedule-details\">";
    html += "<div class=\"schedule-time\">" + timeStr + "</div>";
    html += "<div class=\"schedule-days\">" + daysStr + "</div>";
    html += "<div class=\"schedule-action\">Action: " + actionStr + "</div>";
    html += "</div>";
    html += "<button class=\"btn-delete\" onclick=\"deleteSchedule(" + String(i) + ")\">DELETE</button>";
    html += "</div>";
  }
  
  if (scheduleCount == 0) {
    html = "<div style=\"text-align:center;color:#666;padding:20px;\">No schedules configured</div>";
  }
  
  return html;
}

void handleControl() {
  if (!server.hasArg("action")) {
    server.send(400, "text/plain", "No action specified");
    return;
  }

  String action = server.arg("action");
  Serial.println("Command received: " + action);

  if (action == "stop") {
    motorRunning = false;
    positionLost = true;  // Mark position as unreliable
    Serial.println("EMERGENCY STOP - Position may be inaccurate");
    Serial.println("Use HOME or OPEN to recalibrate position");
  }
  else if (action == "home") {
    // Stop any running operation first
    if (motorRunning) {
      Serial.println("Stopping current operation before homing...");
      motorRunning = false;
      delay(200); // Give time for operation to stop
    }
    Serial.println("Starting homing sequence…");
    autoHome();
    positionLost = false;  // Position is now accurate
  }
  else if (action == "manual" && server.hasArg("direction")) {
    // Check if motor is already running
    if (motorRunning) {
      Serial.println("Motor already running - ignoring manual command (likely homing in progress)");
      server.send(200, "text/plain", "Motor busy - homing in progress");
      return;
    }
    
    String direction = server.arg("direction");
    Serial.println("Manual move: " + direction);

    // Choose speed based on current position
    int speedDelay = NORMAL_SPEED_DELAY;
    if (currentPosition >= (closedPosition * 0.8) && direction == "up") {
      speedDelay = HEAVY_LOAD_DELAY;
      Serial.println("Using HEAVY LOAD speed for manual UP from closed position");
    }

    if (direction == "up") {
      digitalWrite(DIR_PIN, HIGH);
      runMotorSteps(MANUAL_STEP_SIZE, speedDelay, true, true);
      currentPosition = max(0, currentPosition - MANUAL_STEP_SIZE);
    } else if (direction == "down") {
      digitalWrite(DIR_PIN, LOW);
      runMotorSteps(MANUAL_STEP_SIZE, NORMAL_SPEED_DELAY, false, false);
      currentPosition += MANUAL_STEP_SIZE;
    }
  }
  else if (action == "save") {
    if (currentPosition > 50) { // Prevent saving home as closed position
      closedPosition = currentPosition;
      EEPROM.put(EEPROM_ADDRESS, closedPosition);
      EEPROM.commit();
      Serial.println("Saved closed position: " + String(closedPosition) + " steps");
    } else {
      Serial.println("ERROR: Cannot save closed position - too close to home!");
    }
  }
  else if (action == "open") {
    // Check if motor is already running
    if (motorRunning) {
      Serial.println("Motor already running - ignoring open command (likely homing in progress)");
      server.send(200, "text/plain", "Motor busy - homing in progress");
      return;
    }
    Serial.println("Opening curtain - using reliable homing to sensor");
    autoHome(); // Use the SAME reliable function as "Go to Home"
    positionLost = false;  // Position is now accurate
  }
  else if (action == "close" && closedPosition > 0) {
    // Check if motor is already running
    if (motorRunning) {
      Serial.println("Motor already running - ignoring close command (likely homing in progress)");
      server.send(200, "text/plain", "Motor busy - homing in progress");
      return;
    }
    if (positionLost) {
      Serial.println("Cannot close - position lost after emergency stop");
      Serial.println("Please use OPEN (to recalibrate) or HOME first");
    } else {
      Serial.println("Closing curtain - going to closed position");
      moveToPosition(closedPosition); // Go to closed position
    }
  }
  else if (action == "add_schedule") {
    // Add new schedule
    if (scheduleCount >= MAX_SCHEDULES) {
      server.send(400, "text/plain", "Maximum schedules reached");
      return;
    }
    
    if (!server.hasArg("hour") || !server.hasArg("minute") || !server.hasArg("ampm") || 
        !server.hasArg("days") || !server.hasArg("action")) {
      server.send(400, "text/plain", "Missing schedule parameters");
      return;
    }
    
    Schedule newSchedule;
    newSchedule.enabled = true;
    newSchedule.hour = server.arg("hour").toInt();
    newSchedule.minute = server.arg("minute").toInt();
    newSchedule.isPM = (server.arg("ampm") == "PM");
    
    // Parse days (comma-separated day numbers)
    String daysStr = server.arg("days");
    newSchedule.days = 0;
    int start = 0;
    int end = daysStr.indexOf(',');
    
    while (end != -1) {
      int day = daysStr.substring(start, end).toInt();
      if (day >= 0 && day <= 6) {
        newSchedule.days |= (1 << day);
      }
      start = end + 1;
      end = daysStr.indexOf(',', start);
    }
    
    // Handle last day
    int day = daysStr.substring(start).toInt();
    if (day >= 0 && day <= 6) {
      newSchedule.days |= (1 << day);
    }
    
    newSchedule.action = server.arg("action");
    
    schedules[scheduleCount] = newSchedule;
    scheduleCount++;
    
    saveSchedulesToEEPROM();
    
    Serial.println("Added schedule: " + String(newSchedule.hour) + ":" + 
                   String(newSchedule.minute) + " " + (newSchedule.isPM ? "PM" : "AM") + 
                   " - " + newSchedule.action);
  }
  else if (action == "delete_schedule" && server.hasArg("index")) {
    int index = server.arg("index").toInt();
    if (index >= 0 && index < scheduleCount) {
      // Shift schedules down
      for (int i = index; i < scheduleCount - 1; i++) {
        schedules[i] = schedules[i + 1];
      }
      scheduleCount--;
      saveSchedulesToEEPROM();
      
      Serial.println("Deleted schedule at index " + String(index));
    }
  }
  else {
    Serial.println("Unknown or invalid command: " + action);
  }

  server.send(200, "text/plain", "OK");
}

void saveSchedulesToEEPROM() {
  EEPROM.put(SCHEDULE_EEPROM_ADDRESS, scheduleCount);
  EEPROM.put(SCHEDULE_EEPROM_ADDRESS + sizeof(int), schedules);
  EEPROM.commit();
  Serial.println("Saved " + String(scheduleCount) + " schedules to EEPROM");
}

void loadSchedulesFromEEPROM() {
  EEPROM.get(SCHEDULE_EEPROM_ADDRESS, scheduleCount);
  
  if (scheduleCount < 0 || scheduleCount > MAX_SCHEDULES) {
    scheduleCount = 0; // Reset if invalid data
    Serial.println("No valid schedules found in EEPROM");
    return;
  }
  
  EEPROM.get(SCHEDULE_EEPROM_ADDRESS + sizeof(int), schedules);
  Serial.println("Loaded " + String(scheduleCount) + " schedules from EEPROM");
  
  // Print loaded schedules for debugging
  for (int i = 0; i < scheduleCount; i++) {
    Serial.printf("Schedule %d: %02d:%02d %s, Days: %d, Action: %s\n",
                 i, schedules[i].hour, schedules[i].minute,
                 schedules[i].isPM ? "PM" : "AM",
                 schedules[i].days, schedules[i].action.c_str());
  }
}

void moveToPosition(int targetPosition) {
  if (targetPosition == currentPosition) {
    Serial.println("Already at target position");
    return;
  }

  int stepsToMove = abs(targetPosition - currentPosition);
  bool moveUp = (targetPosition < currentPosition);

  Serial.println("Moving to position " + String(targetPosition) + " (currently at " + String(currentPosition) + ")");
  Serial.println("Steps to move: " + String(stepsToMove) + ", Direction: " + String(moveUp ? "UP" : "DOWN"));

  // Use slower speed if starting from closed position (heavy load)
  int speedDelay = NORMAL_SPEED_DELAY;
  if (currentPosition >= (closedPosition * 0.8) && moveUp) {
    speedDelay = HEAVY_LOAD_DELAY;
    Serial.println("Using HEAVY LOAD speed (extra slow) - starting from closed position");
  }

  digitalWrite(DIR_PIN, moveUp ? HIGH : LOW);
  runMotorSteps(stepsToMove, speedDelay, moveUp, moveUp);

  currentPosition = targetPosition;
  Serial.println("Movement complete. New position: " + String(currentPosition));
}

void runMotorSteps(int steps, int delayMicros, bool direction, bool checkSensor) {
  digitalWrite(DIR_PIN, direction ? HIGH : LOW);
  motorRunning = true;

  Serial.println("Running " + String(steps) + " steps, direction: " + String(direction ? "UP" : "DOWN"));
  Serial.println("Speed: " + String(delayMicros) + " microseconds delay");

  int consecutiveHighReadings = 0; // For sensor filtering
  const int SENSOR_CONFIRM_COUNT = 3; // Need 3 consecutive readings for manual moves

  for(int i = 0; i < steps && motorRunning; i++) {
    // Check if stop was requested more frequently for faster response
    if (i % 10 == 0) {  // Check every 10 steps instead of 100
      server.handleClient();
      if (!motorRunning) {
        Serial.println("Emergency stop detected during movement");
        break;
      }
    }

    // Check home sensor if moving up with filtering (sensor triggered = HIGH)
    if (checkSensor && direction) { // Only when moving up
      bool sensorReading = digitalRead(HOME_SENSOR_PIN);
      
      if (sensorReading == HIGH) {
        consecutiveHighReadings++;
        if (consecutiveHighReadings >= SENSOR_CONFIRM_COUNT) {
          Serial.println("Home sensor confirmed during movement - stopping");
          Serial.println("Consecutive HIGH readings: " + String(consecutiveHighReadings));
          currentPosition = 0;
          positionLost = false;  // Position is accurate at home
          break;
        }
      } else {
        consecutiveHighReadings = 0; // Reset on LOW reading
      }
      
      // Debug sensor every 100 steps when checking
      if (i % 100 == 0) {
        Serial.println("Step " + String(i) + ", Sensor: " + String(sensorReading ? "HIGH" : "LOW") + ", Consecutive: " + String(consecutiveHighReadings));
      }
    }

    digitalWrite(STEP_PIN, HIGH);
    delayMicroseconds(50);
    digitalWrite(STEP_PIN, LOW);
    delayMicroseconds(delayMicros);
  }

  motorRunning = false;
  Serial.println("Motor movement finished");
}
